Convertendo as datas do data frame

cidadesBrasil$dt <- as.POSIXct(cidadesBrasil$dt,format='%Y-%m-%d')
cidadesBrasil$Month <- month(cidadesBrasil$dt)
cidadesBrasil$Year <- year(cidadesBrasil$dt)

Carregando os subsets para cada cidade do data frame Palmas

Construindo os Plots
Palmas

p_palmas <- ggplot(palmas, aes(x = (Month), y = AverageTemperature, color = as.factor(Year))) +
  geom_smooth(se = FALSE,fill = NA, size = 2) +
  theme_light(base_size = 20) +
  xlab("M√™s")+
  ylab("Temperatura") +
  scale_color_discrete("") +
  ggtitle("Temperatura m√©dia ao longo dos anos em Palmas") +
  theme(plot.title = element_text(size = 18))
